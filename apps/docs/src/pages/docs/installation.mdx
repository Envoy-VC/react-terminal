# Installation

To install `react-terminal` in your project, run:

```sh npm2yarn
npm install @envoy1084/react-terminal
```

import { Callout } from 'nextra/components'

  ```js {3,4,5} filename="next.config.js"
const config = {
    // other configuration
    experimental: {

export default config;
```
